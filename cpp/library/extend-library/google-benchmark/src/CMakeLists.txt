function(add_bench exe_name)
  add_executable(${exe_name} ${exe_name}.cc)
  target_link_libraries(${exe_name} benchmark::benchmark_main)
endfunction()

add_bench(basic_bench)
add_bench(memcpy_bench)
add_bench(map_bench)
add_bench(intset_bench)
add_bench(memcpy_by_sparse_range_bench)
add_bench(intset_apply_bench)
